# 通过队列和栈实现深度优先遍历和广度优先遍历

深度优先：

Dfs1（node,res=[]）{
    // 递归
    res.push(node);
    if(node.children){
        for (let i = 0; i < node.children.length; i++) {
            Dfs1(node.children[i],res);
          }
    }
    return res
}

 Dfs2(node) {
// 深度用栈，先进后出
let stack = [node];
let res = [];
while (stack.length) {
  let item = stack.pop();
  res.push(item);
  for (let i = item.children.length - 1; i > 0; i--) {
    stack.push(item.children[i]);
  }
}
return stack;
}

广度优先：

  Bfs(node) {
    // 队列先进先出
    let queue = [node];
    let res = [];
    while (queue.length) {
      let item = queue.shift();
      res.push(item);
      for (let i = 0; i < item.children.length; i++) {
        queue.push(item.children[i]);
      }
    }
    return queue;
  }
